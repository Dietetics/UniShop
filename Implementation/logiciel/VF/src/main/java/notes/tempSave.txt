import java.io.*;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class Database {

    public static void refresh() {
        String baseFolderPath = "src/main/resources/data/acheteur"; // Remplacez par votre chemin réel
        String outputFile = "src/main/resources/data/main.csv"; // Remplacez par votre chemin réel

        List<String> orderOfFiles = Arrays.asList("suiviPar", "suivre", "main");
        List<String> allLines = new ArrayList<>();

        File baseFolder = new File(baseFolderPath);
        File[] subFolders = baseFolder.listFiles(File::isDirectory);

        if (subFolders != null) {
            for (File subFolder : subFolders) {
                for (String fileName : orderOfFiles) {
                    File csvFile = new File(subFolder, fileName + ".csv");

                    if (csvFile.exists()) {
                        try (BufferedReader reader = new BufferedReader(new FileReader(csvFile))) {
                            String line;
                            while ((line = reader.readLine()) != null) {
                                allLines.add(line);
                            }
                        } catch (IOException e) {
                            e.printStackTrace();
                        }
                    }
                }
            }
        }

        // Écrire le contenu dans le fichier de sortie
        try (PrintWriter writer = new PrintWriter(new FileWriter(outputFile))) {
            for (String line : allLines) {
                writer.println(line);
            }
        } catch (IOException e) {
            e.printStackTrace();
        }

        System.out.println("Refresh complete.");
    }

    public static void main(String[] args) {
        refresh();
    }
}
